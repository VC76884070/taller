-- =====================================================
-- 3. VISTA PARA ESTADO DE BAHÍAS
-- =====================================================

CREATE OR REPLACE VIEW vista_bahias AS
SELECT 
    b.bahia_numero,
    CASE 
        WHEN p.id IS NOT NULL AND p.fecha_hora_fin_real IS NULL AND p.fecha_hora_inicio_real IS NOT NULL THEN 'ocupado'
        WHEN p.id IS NOT NULL AND p.fecha_hora_fin_real IS NULL THEN 'reservado'
        ELSE 'libre'
    END AS estado,
    o.codigo_unico AS orden_codigo,
    o.estado_global AS orden_estado,
    p.fecha_hora_inicio_estimado,
    p.fecha_hora_fin_estimado
FROM (
    SELECT generate_series(1, 12) AS bahia_numero
) b
LEFT JOIN planificacion p ON p.bahia_asignada = b.bahia_numero AND p.fecha_hora_fin_real IS NULL
LEFT JOIN ordentrabajo o ON o.id = p.id_orden_trabajo;

-- =====================================================
-- 4. VISTA PARA CARGA DE TÉCNICOS
-- =====================================================

CREATE OR REPLACE VIEW vista_tecnicos_carga AS
SELECT 
    u.id,
    u.nombre,
    u.contacto,
    COUNT(a.id_orden_trabajo) FILTER (WHERE a.fecha_hora_final IS NULL) AS ordenes_activas,
    2 AS max_vehiculos
FROM usuario u
LEFT JOIN asignaciontecnico a ON a.id_tecnico = u.id
WHERE u.id_rol = 4
GROUP BY u.id, u.nombre, u.contacto;

-- =====================================================
-- 5. VERIFICAR
-- =====================================================

SELECT * FROM vista_bahias ORDER BY bahia_numero;
SELECT * FROM vista_tecnicos_carga ORDER BY ordenes_activas DESC;

-- =====================================================

-- 1. Agregar campos de tiempos reales a planificacion (si no existen)
ALTER TABLE planificacion 
ADD COLUMN IF NOT EXISTS fecha_hora_inicio_real TIMESTAMP,
ADD COLUMN IF NOT EXISTS fecha_hora_fin_real TIMESTAMP;

-- 2. Vista para saber qué bahías están ocupadas en tiempo real
CREATE OR REPLACE VIEW vista_bahias_ocupadas AS
SELECT 
    p.bahia_asignada,
    p.id_orden_trabajo,
    o.codigo_unico,
    o.estado_global,
    p.fecha_hora_inicio_estimado,
    p.fecha_hora_fin_estimado,
    p.fecha_hora_inicio_real,
    -- Una bahía está ocupada si tiene inicio real y no tiene fin real
    CASE 
        WHEN p.fecha_hora_inicio_real IS NOT NULL AND p.fecha_hora_fin_real IS NULL THEN true
        ELSE false
    END AS esta_ocupada
FROM planificacion p
JOIN ordentrabajo o ON o.id = p.id_orden_trabajo
WHERE p.fecha_hora_fin_real IS NULL;  -- Solo planificaciones activas

-- 3. Función para verificar disponibilidad de bahía en un horario
CREATE OR REPLACE FUNCTION bahia_disponible_en_horario(
    p_bahia INTEGER,
    p_fecha_inicio TIMESTAMP,
    p_fecha_fin TIMESTAMP,
    p_excluir_orden INTEGER DEFAULT NULL
)
RETURNS BOOLEAN AS $$
DECLARE
    v_conflicto INTEGER;
BEGIN
    -- Buscar si hay alguna planificación que ocupe la bahía en ese horario
    SELECT p.id INTO v_conflicto
    FROM planificacion p
    WHERE p.bahia_asignada = p_bahia
    AND p.fecha_hora_fin_real IS NULL  -- Bahía no liberada
    AND (p_excluir_orden IS NULL OR p.id_orden_trabajo != p_excluir_orden)
    AND p.fecha_hora_inicio_estimado < p_fecha_fin
    AND p.fecha_hora_fin_estimado > p_fecha_inicio
    LIMIT 1;
    
    RETURN v_conflicto IS NULL;
END;
$$ LANGUAGE plpgsql;

-- =====================================================
-- 8. CREAR VISTA PARA VER DIAGNÓSTICO CON SERVICIOS
-- =====================================================

CREATE OR REPLACE VIEW vista_diagnostico_completo AS
SELECT 
    dt.id AS diagnostico_id,
    dt.id_orden_trabajo,
    dt.id_tecnico,
    u.nombre AS tecnico_nombre,
    dt.informe,
    dt.url_grabacion_informe,
    dt.transcripcion_informe,
    dt.estado,
    dt.version,
    dt.es_borrador,
    dt.fecha_envio,
    dt.fecha_modificacion,
    -- Servicios asociados
    COALESCE(
        (SELECT json_agg(
            json_build_object(
                'id', st.id,
                'descripcion', st.descripcion,
                'orden', st.orden,
                'precio_estimado', sp.precio_estimado,
                'precio_final', sp.precio_final
            ) ORDER BY st.orden
        ) FROM servicio_tecnico st 
        LEFT JOIN servicio_precio sp ON sp.id_servicio = st.id
        WHERE st.id_diagnostico_tecnico = dt.id),
        '[]'::json
    ) AS servicios,
    -- Contador de fotos
    COALESCE(
        (SELECT COUNT(*) FROM foto_diagnostico fd WHERE fd.id_diagnostico_tecnico = dt.id),
        0
    ) AS total_fotos
FROM diagnostico_tecnico dt
JOIN usuario u ON u.id = dt.id_tecnico;

-- =====================================================
-- 9. FUNCIÓN PARA VERIFICAR LÍMITE DE FOTOS
-- =====================================================

CREATE OR REPLACE FUNCTION verificar_limite_fotos(p_diagnostico_id INTEGER)
RETURNS TABLE(
    puede_subir BOOLEAN,
    fotos_actuales INTEGER,
    max_fotos INTEGER
) AS $$
DECLARE
    v_max_fotos INTEGER;
    v_fotos_actuales INTEGER;
BEGIN
    -- Obtener límite máximo de fotos del diagnóstico
    SELECT COALESCE(dt.max_fotos, 2) INTO v_max_fotos
    FROM diagnostico_tecnico dt
    WHERE dt.id = p_diagnostico_id;
    
    -- Contar fotos actuales
    SELECT COUNT(*) INTO v_fotos_actuales
    FROM foto_diagnostico
    WHERE id_diagnostico_tecnico = p_diagnostico_id;
    
    RETURN QUERY SELECT 
        (v_fotos_actuales < v_max_fotos) AS puede_subir,
        v_fotos_actuales AS fotos_actuales,
        v_max_fotos AS max_fotos;
END;
$$ LANGUAGE plpgsql;

-- =====================================================
-- 10. FUNCIÓN PARA NOTIFICAR APROBACIÓN/RECHAZO
-- =====================================================

CREATE OR REPLACE FUNCTION notificar_tecnico_aprobacion()
RETURNS TRIGGER AS $$
BEGIN
    -- Insertar notificación para el técnico cuando cambia el estado
    IF NEW.estado = 'aprobado' AND OLD.estado != 'aprobado' THEN
        INSERT INTO notificacion (id_usuario_destino, tipo, mensaje, fecha_envio, leida)
        SELECT 
            dt.id_tecnico,
            'diagnostico_aprobado',
            '✅ Tu diagnóstico ha sido APROBADO. El Jefe Operativo asignará los precios próximamente.',
            NOW(),
            FALSE
        FROM diagnostico_tecnico dt
        WHERE dt.id = NEW.id;
    
    ELSIF NEW.estado = 'rechazado' AND OLD.estado != 'rechazado' THEN
        INSERT INTO notificacion (id_usuario_destino, tipo, mensaje, fecha_envio, leida)
        SELECT 
            dt.id_tecnico,
            'diagnostico_rechazado',
            '❌ Tu diagnóstico ha sido RECHAZADO. Revisa las observaciones del Jefe de Taller y realiza las correcciones necesarias.',
            NOW(),
            FALSE
        FROM diagnostico_tecnico dt
        WHERE dt.id = NEW.id;
    END IF;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trigger_notificar_tecnico ON diagnostico_tecnico;
CREATE TRIGGER trigger_notificar_tecnico
    AFTER UPDATE OF estado ON diagnostico_tecnico
    FOR EACH ROW
    EXECUTE FUNCTION notificar_tecnico_aprobacion();

-- 11. FUNCIÓN PARA ACTUALIZAR ESTADO DE ORDEN CUANDO SE APRUEBA DIAGNÓSTICO
-- =====================================================

CREATE OR REPLACE FUNCTION actualizar_estado_orden_aprobacion()
RETURNS TRIGGER AS $$
BEGIN
    -- Cuando el diagnóstico es aprobado, cambiar estado de la orden a 'PendienteAprobacion'
    -- para que el Jefe Operativo asigne precios
    IF NEW.estado = 'aprobado' AND OLD.estado != 'aprobado' THEN
        UPDATE ordentrabajo 
        SET estado_global = 'PendienteAprobacion'
        WHERE id = NEW.id_orden_trabajo;
        
        -- Notificar al Jefe Operativo que debe asignar precios
        INSERT INTO notificacion (id_usuario_destino, tipo, mensaje, fecha_envio, leida)
        SELECT 
            ot.id_jefe_operativo,
            'precios_pendientes',
            '💰 Se requiere asignación de precios para la orden ' || ot.codigo_unico,
            NOW(),
            FALSE
        FROM ordentrabajo ot
        WHERE ot.id = NEW.id_orden_trabajo;
    END IF;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trigger_actualizar_estado_orden ON diagnostico_tecnico;
CREATE TRIGGER trigger_actualizar_estado_orden
    AFTER UPDATE OF estado ON diagnostico_tecnico
    FOR EACH ROW
    EXECUTE FUNCTION actualizar_estado_orden_aprobacion();

-- Recrear la vista con la nueva estructura
CREATE OR REPLACE VIEW vista_diagnostico_completo AS
SELECT 
    dt.id AS diagnostico_id,
    dt.id_orden_trabajo,
    dt.id_tecnico,
    u.nombre AS tecnico_nombre,
    dt.informe,
    dt.url_grabacion_informe,
    dt.transcripcion_informe,
    dt.estado,
    dt.version,
    dt.es_borrador,
    dt.fecha_envio,
    dt.fecha_modificacion,
    -- Servicios asociados con sus precios
    COALESCE(
        (SELECT json_agg(
            json_build_object(
                'id', st.id,
                'descripcion', st.descripcion,
                'orden', st.orden,
                'precio_estimado', sp.precio_estimado,
                'precio_final', sp.precio_final,
                'tiene_solicitud_repuesto', EXISTS(
                    SELECT 1 FROM solicitud_cotizacion_repuesto scr 
                    WHERE scr.id_servicio = st.id
                )
            ) ORDER BY st.orden
        ) FROM servicio_tecnico st 
        LEFT JOIN servicio_precio sp ON sp.id_servicio = st.id
        WHERE st.id_diagnostico_tecnico = dt.id),
        '[]'::json
    ) AS servicios,
    -- Solicitudes de repuestos activas
    COALESCE(
        (SELECT json_agg(
            json_build_object(
                'id', scr.id,
                'descripcion_pieza', scr.descripcion_pieza,
                'cantidad', scr.cantidad,
                'estado', scr.estado,
                'precio_cotizado', scr.precio_cotizado,
                'fecha_solicitud', scr.fecha_solicitud
            )
        ) FROM solicitud_cotizacion_repuesto scr
        WHERE scr.id_orden_trabajo = dt.id_orden_trabajo AND scr.estado != 'rechazado'),
        '[]'::json
    ) AS solicitudes_repuestos,
    -- Contador de fotos
    COALESCE(
        (SELECT COUNT(*) FROM foto_diagnostico fd WHERE fd.id_diagnostico_tecnico = dt.id),
        0
    ) AS total_fotos
FROM diagnostico_tecnico dt
JOIN usuario u ON u.id = dt.id_tecnico;
