-- =====================================================
-- SCRIPT PARA CREAR BASE DE DATOS EN POSTGRESQL (SUPABASE)
-- BASADO EN EL DIAGRAMA ENTIDAD-RELACIÓN PROPORCIONADO
-- CON CORRECCIONES:
-- 1. id_filtro agregado a Proveedor como FK
-- 2. url_foto_lateral → url_lateral_izquierda + url_lateral_derecha en Recepcion
-- 3. url_observacion_obs → url_grabacion_observacion en ObservacionDiagnostico
-- =====================================================

-- =====================================================
-- TABLAS PRINCIPALES (SIN DEPENDENCIAS EXTERNAS)
-- =====================================================

-- Tabla Rol
CREATE TABLE Rol (
    id SERIAL PRIMARY KEY,
    nombre_rol VARCHAR(100) NOT NULL,
    descripcion TEXT
);

-- Tabla Proveedor
CREATE TABLE Proveedor (
    id SERIAL PRIMARY KEY,
    id_filtro INTEGER,
    nombre VARCHAR(200) NOT NULL,
    telefono VARCHAR(50),
    ubicacion_gps VARCHAR(255),
    propietario VARCHAR(200)
);

-- Tabla Cliente
CREATE TABLE Cliente (
    id SERIAL PRIMARY KEY,
    id_usuario INTEGER,
    tipo_documento VARCHAR(50),
    numero_documento VARCHAR(50) UNIQUE
);

-- Tabla Usuario (depende de Rol)
CREATE TABLE Usuario (
    id SERIAL PRIMARY KEY,
    id_rol INTEGER NOT NULL,
    nombre VARCHAR(200) NOT NULL,
    contacto VARCHAR(100),
    ubicacion VARCHAR(255),
    contrasenia VARCHAR(255) NOT NULL,
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_rol) REFERENCES Rol(id)
);

-- Tabla Filtro_herramienta
CREATE TABLE Filtro_herramienta (
    id SERIAL PRIMARY KEY,
    nombre_filtro VARCHAR(100) NOT NULL,
    descripcion TEXT
);

-- Tabla Repuesto
CREATE TABLE Repuesto (
    id SERIAL PRIMARY KEY,
    codigo VARCHAR(50) UNIQUE NOT NULL,
    nombre VARCHAR(200) NOT NULL,
    descripcion TEXT,
    cantidad_stock INTEGER DEFAULT 0,
    precio_unitario DECIMAL(10, 2)
);

-- Tabla Servicio
CREATE TABLE Servicio (
    id SERIAL PRIMARY KEY,
    id_diagnostico_tecnico INTEGER,
    descripcion TEXT NOT NULL,
    precio_estimado DECIMAL(10, 2),
    precio_final DECIMAL(10, 2)
);

-- =====================================================
-- TABLAS CON DEPENDENCIAS
-- =====================================================

-- Tabla Vehiculo (depende de Cliente)
CREATE TABLE Vehiculo (
    id SERIAL PRIMARY KEY,
    id_cliente INTEGER NOT NULL,
    placa VARCHAR(20) UNIQUE NOT NULL,
    marca VARCHAR(100),
    modelo VARCHAR(100),
    anio INTEGER,
    kilometraje INTEGER,
    FOREIGN KEY (id_cliente) REFERENCES Cliente(id)
);

-- Tabla CarritoHerramientas (depende de Usuario como técnico)
CREATE TABLE CarritoHerramientas (
    id SERIAL PRIMARY KEY,
    id_tecnico INTEGER NOT NULL,
    nombre VARCHAR(200) NOT NULL,
    codigo VARCHAR(50) UNIQUE NOT NULL,
    estado VARCHAR(50),
    FOREIGN KEY (id_tecnico) REFERENCES Usuario(id)
);

-- Tabla Herramienta (depende de CarritoHerramientas)
CREATE TABLE Herramienta (
    id SERIAL PRIMARY KEY,
    id_carritoherramienta INTEGER NOT NULL,
    nombre VARCHAR(200) NOT NULL,
    descripcion VARCHAR(255),
    tipo VARCHAR(100),
    marca VARCHAR(100),
    FOREIGN KEY (id_carritoherramienta) REFERENCES CarritoHerramientas(id)
);

-- Tabla ControlCarrito (depende de CarritoHerramientas)
CREATE TABLE ControlCarrito (
    id SERIAL PRIMARY KEY,
    id_carritoherramienta INTEGER NOT NULL,
    fecha_control TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    estado_encontrado VARCHAR(100),
    observacion TEXT,
    url_evidencia TEXT,
    FOREIGN KEY (id_carritoherramienta) REFERENCES CarritoHerramientas(id)
);

-- Tabla ordenTrabajo (depende de Vehiculo y Usuario)
CREATE TABLE ordenTrabajo (
    id SERIAL PRIMARY KEY,
    codigo_unico VARCHAR(100) UNIQUE NOT NULL,
    id_vehiculo INTEGER NOT NULL,
    id_jefe_operativo INTEGER NOT NULL,
    fecha_ingreso TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_salida TIMESTAMP,
    estado_global VARCHAR(50) DEFAULT 'EnRecepcion',
    FOREIGN KEY (id_vehiculo) REFERENCES Vehiculo(id),
    FOREIGN KEY (id_jefe_operativo) REFERENCES Usuario(id)
);

-- Tabla Recepcion (depende de ordenTrabajo) - CORREGIDA
CREATE TABLE Recepcion (
    id SERIAL PRIMARY KEY,
    id_orden_trabajo INTEGER NOT NULL UNIQUE,
    url_lateral_izquierda TEXT,
    url_lateral_derecha TEXT,
    url_foto_frontal TEXT,
    url_foto_trasera TEXT,
    url_foto_superior TEXT,
    url_foto_inferior TEXT,
    url_foto_tablero TEXT,
    url_grabacion_problema TEXT,
    transcripcion_problema TEXT,
    FOREIGN KEY (id_orden_trabajo) REFERENCES ordenTrabajo(id)
);

-- Tabla Diagnostico_tecnico (depende de ordenTrabajo y Usuario)
CREATE TABLE Diagnostico_tecnico (
    id SERIAL PRIMARY KEY,
    id_orden_trabajo INTEGER NOT NULL,
    id_tecnico INTEGER NOT NULL,
    informe TEXT,
    url_grabacion_informe TEXT,
    transcripcion_informe TEXT,
    estado VARCHAR(50),
    fecha_envio TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    version INTEGER DEFAULT 1,
    FOREIGN KEY (id_orden_trabajo) REFERENCES ordenTrabajo(id),
    FOREIGN KEY (id_tecnico) REFERENCES Usuario(id)
);

-- Tabla Foto_diagnostico (depende de Diagnostico_tecnico)
CREATE TABLE Foto_diagnostico (
    id SERIAL PRIMARY KEY,
    id_diagnostico_tecnico INTEGER NOT NULL,
    url_foto TEXT NOT NULL,
    descripcion_tecnico VARCHAR(255),
    FOREIGN KEY (id_diagnostico_tecnico) REFERENCES Diagnostico_tecnico(id)
);

-- Tabla DiagnostigoInicial (depende de ordenTrabajo y Usuario)
CREATE TABLE DiagnostigoInicial (
    id SERIAL PRIMARY KEY,
    id_orden_trabajo INTEGER NOT NULL,
    id_jefe_taller INTEGER NOT NULL,
    diagnostigo TEXT NOT NULL,
    fecha_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_orden_trabajo) REFERENCES ordenTrabajo(id),
    FOREIGN KEY (id_jefe_taller) REFERENCES Usuario(id)
);

-- Tabla ObservacionDiagnostico (depende de Diagnostico_tecnico y Usuario) - CORREGIDA
CREATE TABLE ObservacionDiagnostico (
    id SERIAL PRIMARY KEY,
    id_diagnostico_tecnico INTEGER NOT NULL,
    id_jefe_taller INTEGER NOT NULL,
    observacion TEXT NOT NULL,
    url_grabacion_observacion TEXT,
    transcripcion_obs TEXT,
    fecha_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_diagnostico_tecnico) REFERENCES Diagnostico_tecnico(id),
    FOREIGN KEY (id_jefe_taller) REFERENCES Usuario(id)
);

-- Tabla AsignacionTecnico (depende de ordenTrabajo y Usuario)
CREATE TABLE AsignacionTecnico (
    id SERIAL PRIMARY KEY,
    id_orden_trabajo INTEGER NOT NULL,
    id_tecnico INTEGER NOT NULL,
    fecha_hora_inicio TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_hora_final TIMESTAMP,
    FOREIGN KEY (id_orden_trabajo) REFERENCES ordenTrabajo(id),
    FOREIGN KEY (id_tecnico) REFERENCES Usuario(id)
);

-- Tabla AvanceTrabajo (depende de ordenTrabajo y Usuario)
CREATE TABLE AvanceTrabajo (
    id SERIAL PRIMARY KEY,
    id_orden_trabajo INTEGER NOT NULL,
    id_tecnico INTEGER NOT NULL,
    fecha_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    descripcion TEXT,
    tipo_avance VARCHAR(100),
    estado_asociado VARCHAR(100),
    FOREIGN KEY (id_orden_trabajo) REFERENCES ordenTrabajo(id),
    FOREIGN KEY (id_tecnico) REFERENCES Usuario(id)
);

-- Tabla Foto_avance (depende de AvanceTrabajo)
CREATE TABLE Foto_avance (
    id SERIAL PRIMARY KEY,
    id_avance INTEGER NOT NULL,
    url_foto TEXT NOT NULL,
    descripcion_tecnico VARCHAR(255),
    fecha_subida TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_avance) REFERENCES AvanceTrabajo(id)
);

-- Tabla SeguimientoOrden (depende de ordenTrabajo)
CREATE TABLE SeguimientoOrden (
    id SERIAL PRIMARY KEY,
    id_orden_trabajo INTEGER NOT NULL,
    estado VARCHAR(50),
    motivo_pausa TEXT,
    fecha_hora_cambio TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    notificaciones_enviadas INTEGER DEFAULT 0,
    FOREIGN KEY (id_orden_trabajo) REFERENCES ordenTrabajo(id)
);

-- Tabla Planificacion (depende de ordenTrabajo)
CREATE TABLE Planificacion (
    id SERIAL PRIMARY KEY,
    id_orden_trabajo INTEGER NOT NULL UNIQUE,
    bahia_asignada INTEGER,
    horas_estimadas DECIMAL(5, 2),
    fecha_hora_inicio_estimado TIMESTAMP,
    fecha_hora_fin_estimado TIMESTAMP,
    fecha_hora_inicio_real TIMESTAMP,
    fecha_hora_fin_real TIMESTAMP,
    FOREIGN KEY (id_orden_trabajo) REFERENCES ordenTrabajo(id)
);

-- Tabla Cotizacion (depende de ordenTrabajo)
CREATE TABLE Cotizacion (
    id SERIAL PRIMARY KEY,
    id_orden_trabajo INTEGER NOT NULL,
    fecha_generacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    estado VARCHAR(50) DEFAULT 'Enviada',
    FOREIGN KEY (id_orden_trabajo) REFERENCES ordenTrabajo(id)
);

-- Tabla CotizacionDetalle (depende de Cotizacion y Servicio)
CREATE TABLE CotizacionDetalle (
    id SERIAL PRIMARY KEY,
    id_cotizacion INTEGER NOT NULL,
    id_servicio INTEGER,
    servicio_descripcion TEXT,
    precio DECIMAL(10, 2) NOT NULL,
    aprobado_por_cliente BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (id_cotizacion) REFERENCES Cotizacion(id),
    FOREIGN KEY (id_servicio) REFERENCES Servicio(id)
);

-- Tabla CompraRepuesto (depende de Proveedor y Usuario)
CREATE TABLE CompraRepuesto (
    id SERIAL PRIMARY KEY,
    id_proveedor INTEGER NOT NULL,
    id_encargado_repuestos INTEGER NOT NULL,
    fecha_compra TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    total DECIMAL(10, 2),
    url_foto_comprobante TEXT,
    FOREIGN KEY (id_proveedor) REFERENCES Proveedor(id),
    FOREIGN KEY (id_encargado_repuestos) REFERENCES Usuario(id)
);

-- Tabla CompraDetalle (depende de CompraRepuesto y Repuesto)
CREATE TABLE CompraDetalle (
    id SERIAL PRIMARY KEY,
    id_compra INTEGER NOT NULL,
    id_repuesto INTEGER NOT NULL,
    cantidad INTEGER NOT NULL,
    precio_compra DECIMAL(10, 2) NOT NULL,
    url_foto_repuesto TEXT,
    FOREIGN KEY (id_compra) REFERENCES CompraRepuesto(id),
    FOREIGN KEY (id_repuesto) REFERENCES Repuesto(id)
);

-- Tabla UsoRepuesto (depende de ordenTrabajo y Repuesto)
CREATE TABLE UsoRepuesto (
    id SERIAL PRIMARY KEY,
    id_orden_trabajo INTEGER NOT NULL,
    id_repuesto INTEGER NOT NULL,
    cantidad INTEGER NOT NULL,
    fecha_uso TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_orden_trabajo) REFERENCES ordenTrabajo(id),
    FOREIGN KEY (id_repuesto) REFERENCES Repuesto(id)
);

-- Tabla TransaccionFinanciera (depende de ordenTrabajo, CompraRepuesto y Usuario)
CREATE TABLE TransaccionFinanciera (
    id SERIAL PRIMARY KEY,
    id_orden_trabajo INTEGER,
    id_compra INTEGER,
    id_usuario_registra INTEGER NOT NULL,
    tipo VARCHAR(50) NOT NULL,
    monto DECIMAL(10, 2) NOT NULL,
    descripcion TEXT,
    fecha_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_orden_trabajo) REFERENCES ordenTrabajo(id),
    FOREIGN KEY (id_compra) REFERENCES CompraRepuesto(id),
    FOREIGN KEY (id_usuario_registra) REFERENCES Usuario(id)
);

-- Tabla AgendaCliente (depende de Cliente)
CREATE TABLE AgendaCliente (
    id SERIAL PRIMARY KEY,
    id_cliente INTEGER NOT NULL,
    fecha_solicitud TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_agendada TIMESTAMP NOT NULL,
    descripcion_problema TEXT,
    estado VARCHAR(50) DEFAULT 'Pendiente',
    FOREIGN KEY (id_cliente) REFERENCES Cliente(id)
);

-- Tabla Notificacion (depende de Usuario)
CREATE TABLE Notificacion (
    id SERIAL PRIMARY KEY,
    id_usuario_destino INTEGER NOT NULL,
    tipo VARCHAR(100),
    mensaje TEXT,
    fecha_envio TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    leida BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (id_usuario_destino) REFERENCES Usuario(id)
);

-- Tabla RendicionDiaria (depende de Usuario)
CREATE TABLE RendicionDiaria (
    id SERIAL PRIMARY KEY,
    id_encargado_repuestos INTEGER NOT NULL,
    id_jefe_operativo INTEGER NOT NULL,
    fecha DATE DEFAULT CURRENT_DATE,
    monto_total DECIMAL(10, 2),
    observaciones TEXT,
    FOREIGN KEY (id_encargado_repuestos) REFERENCES Usuario(id),
    FOREIGN KEY (id_jefe_operativo) REFERENCES Usuario(id)
);

-- Tabla RendicionSemanal (depende de Usuario)
CREATE TABLE RendicionSemanal (
    id SERIAL PRIMARY KEY,
    id_admin_general INTEGER NOT NULL,
    id_jefe_operativo INTEGER NOT NULL,
    fecha_inicio DATE NOT NULL,
    fecha_fin DATE NOT NULL,
    monto_total DECIMAL(10, 2),
    observaciones TEXT,
    FOREIGN KEY (id_admin_general) REFERENCES Usuario(id),
    FOREIGN KEY (id_jefe_operativo) REFERENCES Usuario(id)
);

    expira TIMESTAMP NOT NULL,

);

-- Tabla para comunicados
CREATE TABLE IF NOT EXISTS comunicado (
    id SERIAL PRIMARY KEY,
    titulo VARCHAR(255) NOT NULL,
    contenido TEXT NOT NULL,
    prioridad VARCHAR(50) DEFAULT 'normal',
    estado VARCHAR(50) DEFAULT 'activo',
    destinatarios JSONB DEFAULT '[]',
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    creado_por INTEGER,
    FOREIGN KEY (creado_por) REFERENCES Usuario(id)
);

-- Índices para mejorar rendimiento
CREATE INDEX IF NOT EXISTS idx_comunicado_estado ON comunicado(estado);
CREATE INDEX IF NOT EXISTS idx_comunicado_prioridad ON comunicado(prioridad);
CREATE INDEX IF NOT EXISTS idx_comunicado_fecha ON comunicado(fecha_creacion DESC);
CREATE INDEX IF NOT EXISTS idx_comunicado_destinatarios ON comunicado USING gin (destinatarios);

-- Comentarios
COMMENT ON TABLE comunicado IS 'Comunicados internos del taller';
COMMENT ON COLUMN comunicado.prioridad IS 'Prioridad del comunicado: normal, importante, urgente';
COMMENT ON COLUMN comunicado.estado IS 'Estado del comunicado: activo, inactivo';
COMMENT ON COLUMN comunicado.destinatarios IS 'Array de roles destinatarios: jefe_operativo, jefe_taller, tecnico, encargado_repuestos, admin_general';

    -- Una bahía está ocupada si tiene inicio real y no tiene fin real
    CASE 
        WHEN p.fecha_hora_inicio_real IS NOT NULL AND p.fecha_hora_fin_real IS NULL THEN true
        ELSE false
    END AS esta_ocupada

WHERE p.fecha_hora_fin_real IS NULL;  -- Solo planificaciones activas

-- 3. Función para verificar disponibilidad de bahía en un horario
CREATE OR REPLACE FUNCTION bahia_disponible_en_horario(
    p_bahia INTEGER,
    p_fecha_inicio TIMESTAMP,
    p_fecha_fin TIMESTAMP,
    p_excluir_orden INTEGER DEFAULT NULL
)
RETURNS BOOLEAN AS $$
DECLARE
    v_conflicto INTEGER;
BEGIN
    -- Buscar si hay alguna planificación que ocupe la bahía en ese horario
    SELECT p.id INTO v_conflicto
    FROM planificacion p
    WHERE p.bahia_asignada = p_bahia
    AND p.fecha_hora_fin_real IS NULL  -- Bahía no liberada
    AND (p_excluir_orden IS NULL OR p.id_orden_trabajo != p_excluir_orden)
    AND p.fecha_hora_inicio_estimado < p_fecha_fin
    AND p.fecha_hora_fin_estimado > p_fecha_inicio
    LIMIT 1;
    
    RETURN v_conflicto IS NULL;
END;
$$ LANGUAGE plpgsql;

-- 9. FUNCIÓN PARA VERIFICAR LÍMITE DE FOTOS
-- =====================================================

CREATE OR REPLACE FUNCTION verificar_limite_fotos(p_diagnostico_id INTEGER)
RETURNS TABLE(
    puede_subir BOOLEAN,
    fotos_actuales INTEGER,
    max_fotos INTEGER
) AS $$
DECLARE
    v_max_fotos INTEGER;
    v_fotos_actuales INTEGER;
BEGIN
    -- Obtener límite máximo de fotos del diagnóstico
    SELECT COALESCE(dt.max_fotos, 2) INTO v_max_fotos
    FROM diagnostico_tecnico dt
    WHERE dt.id = p_diagnostico_id;
    
    -- Contar fotos actuales
    SELECT COUNT(*) INTO v_fotos_actuales
    FROM foto_diagnostico
    WHERE id_diagnostico_tecnico = p_diagnostico_id;
    
    RETURN QUERY SELECT 
        (v_fotos_actuales < v_max_fotos) AS puede_subir,
        v_fotos_actuales AS fotos_actuales,
        v_max_fotos AS max_fotos;
END;
$$ LANGUAGE plpgsql;

-- =====================================================
-- 10. FUNCIÓN PARA NOTIFICAR APROBACIÓN/RECHAZO
-- =====================================================

CREATE OR REPLACE FUNCTION notificar_tecnico_aprobacion()
RETURNS TRIGGER AS $$
BEGIN
    -- Insertar notificación para el técnico cuando cambia el estado
    IF NEW.estado = 'aprobado' AND OLD.estado != 'aprobado' THEN
        INSERT INTO notificacion (id_usuario_destino, tipo, mensaje, fecha_envio, leida)
        SELECT 
            dt.id_tecnico,
            'diagnostico_aprobado',
            '✅ Tu diagnóstico ha sido APROBADO. El Jefe Operativo asignará los precios próximamente.',
            NOW(),
            FALSE
        FROM diagnostico_tecnico dt
        WHERE dt.id = NEW.id;
    
    ELSIF NEW.estado = 'rechazado' AND OLD.estado != 'rechazado' THEN
        INSERT INTO notificacion (id_usuario_destino, tipo, mensaje, fecha_envio, leida)
        SELECT 
            dt.id_tecnico,
            'diagnostico_rechazado',
            '❌ Tu diagnóstico ha sido RECHAZADO. Revisa las observaciones del Jefe de Taller y realiza las correcciones necesarias.',
            NOW(),
            FALSE
        FROM diagnostico_tecnico dt
        WHERE dt.id = NEW.id;
    END IF;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trigger_notificar_tecnico ON diagnostico_tecnico;
CREATE TRIGGER trigger_notificar_tecnico
    AFTER UPDATE OF estado ON diagnostico_tecnico
    FOR EACH ROW
    EXECUTE FUNCTION notificar_tecnico_aprobacion();

-- 11. FUNCIÓN PARA ACTUALIZAR ESTADO DE ORDEN CUANDO SE APRUEBA DIAGNÓSTICO

CREATE OR REPLACE FUNCTION actualizar_estado_orden_aprobacion()
RETURNS TRIGGER AS $$
BEGIN
    -- Cuando el diagnóstico es aprobado, cambiar estado de la orden a 'PendienteAprobacion'
    -- para que el Jefe Operativo asigne precios
    IF NEW.estado = 'aprobado' AND OLD.estado != 'aprobado' THEN
        UPDATE ordentrabajo 
        SET estado_global = 'PendienteAprobacion'
        WHERE id = NEW.id_orden_trabajo;
        
        -- Notificar al Jefe Operativo que debe asignar precios
        INSERT INTO notificacion (id_usuario_destino, tipo, mensaje, fecha_envio, leida)
        SELECT 
            ot.id_jefe_operativo,
            'precios_pendientes',
            '💰 Se requiere asignación de precios para la orden ' || ot.codigo_unico,
            NOW(),
            FALSE
        FROM ordentrabajo ot
        WHERE ot.id = NEW.id_orden_trabajo;
    END IF;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

    AFTER UPDATE OF estado ON diagnostico_tecnico
